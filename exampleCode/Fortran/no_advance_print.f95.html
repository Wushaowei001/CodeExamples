PROGRAM no_advance_print

  ! Non-advancing I/O now has a standard solution.  Note that the FMT is required if ADVANCE is used.
  WRITE (*,FMT="(1a)",ADVANCE='NO') 'The value is: '
  WRITE (*,FMT="(1f5.2)") 1.3

END PROGRAM no_advance_print
