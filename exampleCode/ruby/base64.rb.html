require 'base64'

pText = 'Hello, World!'
puts("pText: '#{pText.inspect}")

cText = Base64.encode64(pText)
puts("cText: '#{cText.inspect}")

pText = Base64.decode64(cText)
puts("pText: '#{pText.inspect}")
