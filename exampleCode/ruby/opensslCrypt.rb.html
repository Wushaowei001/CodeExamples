# @Std       Ruby 1.8

require 'openssl'

pText = "Hello, World!"
aKey  = 'hellohello'

puts("pText: #{pText.inspect}")

cipher = OpenSSL::Cipher::Cipher.new( "des3" )
cipher.encrypt(aKey)
cText = cipher.update(pText)  # repeatedly call this till no more data.
cText << cipher.final

cipher.decrypt(aKey)
pText = cipher.update(cText)
pText << cipher.final

puts("pText: #{pText.inspect}")
