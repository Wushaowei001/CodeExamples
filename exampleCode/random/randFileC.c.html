int newFileRand(char *fileName) {
  int FD;
  return (FD = open(fileName, O_RDONLY));
}

/* ------------------------------------------------------------------------------------------------------------------------------ */
int delFileRand(int FD) {
  return close(FD);
}

char genFileRandCharS(int FD) {
  char retValue;
  if(FD < 0) return 0;
  if(read(FD, &retValue, sizeof(retValue)) < sizeof(retValue)) return 0;
  return retValue;
}
