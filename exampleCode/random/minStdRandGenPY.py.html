#-----------------------------------------------------------------------------------------------------------------------------------
m     = 2147483647;   # 2**(31)-1 for MSLCG
a     = 16807;        # 7**5 for MSLCG
q     = int(m / a);   # 127773 for MSLCG
r     = m % a;        #   2836 for MSLCG
randN = 1;            # Set the seed (first random number)

for x in range(10):
    randN = a * (randN % q) - r * int(randN / q);
    if randN<0 :
        randN += m;
    print '%12d' % randN;
