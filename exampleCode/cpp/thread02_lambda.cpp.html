////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////
#include <thread>
#include <iostream>

////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////
int main(){

  // Construct thread (which will launch workerThread in a new thread)
  std::thread aThread([](){
      std::cout << "Hello from thread!" << std::endl;
    });

  // We "Join" with thread so that the main thread will wait for aThread to finish.
  aThread.join();

  std::cout << "DONE!" << std::endl;

  return 0;
} // end func main
