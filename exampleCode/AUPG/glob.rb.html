dirToRead = ARGV[0]

# Read the directory entries.
dirEnts = Dir.glob("#{dirToRead}/*") + Dir.glob("#{dirToRead}/.*")

# Print out the info
dirEnts.each do |dirEnt| 
  puts("File: #{dirEnt}")
end
